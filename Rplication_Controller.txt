apiVersion: v1
kind: ReplicationController
metadata:
	name: <RC_Name>
	namespace: <Namespace_Name>
spec:
       replicas: <Number of pods you want/by default 1>
       template:  ###This is POD template
		metadata:
		 	name: <pod name>
			labels:
				<key>: <value>
			namespace: <Namespace_Name>
		spec:
			containers:
				- name: <containerName>
				  image: <ImageName>
				  ports:
					- containerPort: <ContainerPort>
